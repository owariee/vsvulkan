17struct DeletionQueue
18{
19 std::deque<std::function<void()>> deletors;
20
21 void push_function(std::function<void()>&& function) {
22 deletors.push_back(function);
23 }
24
25 void flush() {
26 // reverse iterate the deletion queue to execute all the functions
27 for (auto it = deletors.rbegin(); it != deletors.rend(); it++) {
28 (*it)(); //call functors
29 }
30
31 deletors.clear();
32 }
33};
